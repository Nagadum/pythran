#pythran export pi_estimate(int)
def pi_estimate(DARTS):
 hits = 0
 "omp parallel for private(x,y,dist), reduction(+:hits)"
 for i in xrange (0, DARTS):
  x = random()
  y = random()
  dist = sqrt(pow(x, 2) + pow(y, 2))
  if dist <= 1.0:
   hits += 1.0
 pi = 4 * (hits / DARTS)
 return pi
\end{lstlisting}
